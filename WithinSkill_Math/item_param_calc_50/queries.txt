CREATE OR REPLACE TABLE [REDACTED].PRODUCT_TESTING.QUESTIONID_SKILLID_ASSOCIATIONS AS
SELECT DISTINCT
    QUESTIONID,
    ARRAY_COMPACT(
        ARRAY_CAT(
            COALESCE(
                FIRST_VALUE(QUESTIONSKILLS) OVER (
                    PARTITION BY QUESTIONID
                    ORDER BY OCCURREDAT DESC
                ),
                ARRAY_CONSTRUCT()
            ),
            ARRAY_CONSTRUCT(
                FIRST_VALUE(RLSKILLID) OVER (
                    PARTITION BY QUESTIONID
                    ORDER BY OCCURREDAT DESC
                )
            )
        )
    ) AS UNIQUE_SKILLS
FROM PROGRESS.ANSWER_EVENTS
WHERE OCCURREDAT >= '2024-05-01'
ORDER BY QUESTIONID;


CREATE OR REPLACE TABLE [REDACTED].PRODUCT_TESTING.MATH_SIMULATED_MULTIPLE_SKILLS_UPDATE_ETL AS
WITH exploded_skills AS (
    SELECT 
        e.STUDENTID,
        e.QUESTIONID,
        e.CORRECTNESS,
        e.OCCURREDAT,
        e.METADATA:durationSeconds::FLOAT as DURATIONSECONDS,
        e.METADATA:answerPositionInSession::FLOAT as ANSWERPOSITIONINSESSION,
        e.RLSKILLID,
        f.value::STRING AS SKILL
    FROM 
        PROGRESS.ANSWER_EVENTS e
    JOIN 
        [REDACTED].PRODUCT_TESTING.QUESTIONID_SKILLID_ASSOCIATIONS a
    ON 
        e.QUESTIONID = a.QUESTIONID
    ,LATERAL FLATTEN(input => a.UNIQUE_SKILLS) f
    WHERE 
        e.OCCURREDAT >= '2024-05-01'
)
SELECT DISTINCT *
FROM exploded_skills
ORDER BY STUDENTID, SKILL, OCCURREDAT, ANSWERPOSITIONINSESSION;


CREATE OR REPLACE TABLE [REDACTED].PRODUCT_TESTING.SIMULATED_MULTIPLE_SKILLS_UPDATE_ETL_WITH_LAGS AS
WITH ranked_events AS (
    SELECT 
        *,
        ROW_NUMBER() OVER (PARTITION BY STUDENTID, SKILL ORDER BY OCCURREDAT ASC, ANSWERPOSITIONINSESSION ASC) AS event_rank
    FROM [REDACTED].PRODUCT_TESTING.MATH_SIMULATED_MULTIPLE_SKILLS_UPDATE_ETL
),
base_events AS (
    SELECT *
    FROM ranked_events
)
SELECT 
    b.STUDENTID,
    b.SKILL,
    b.QUESTIONID,
    b.OCCURREDAT,
    b.CORRECTNESS,
    b.DURATIONSECONDS,
    b.ANSWERPOSITIONINSESSION,
    b.event_rank,
    -- Past lags
    l1.QUESTIONID AS QUESTIONID_LAG_1,
    l2.QUESTIONID AS QUESTIONID_LAG_2,
    l3.QUESTIONID AS QUESTIONID_LAG_3,
    l4.QUESTIONID AS QUESTIONID_LAG_4,
    l5.QUESTIONID AS QUESTIONID_LAG_5,
    l6.QUESTIONID AS QUESTIONID_LAG_6,
    l7.QUESTIONID AS QUESTIONID_LAG_7,
    l8.QUESTIONID AS QUESTIONID_LAG_8,
    l9.QUESTIONID AS QUESTIONID_LAG_9,
    l10.QUESTIONID AS QUESTIONID_LAG_10,
    l1.OCCURREDAT AS OCCURREDAT_LAG_1,
    l2.OCCURREDAT AS OCCURREDAT_LAG_2,
    l3.OCCURREDAT AS OCCURREDAT_LAG_3,
    l4.OCCURREDAT AS OCCURREDAT_LAG_4,
    l5.OCCURREDAT AS OCCURREDAT_LAG_5,
    l6.OCCURREDAT AS OCCURREDAT_LAG_6,
    l7.OCCURREDAT AS OCCURREDAT_LAG_7,
    l8.OCCURREDAT AS OCCURREDAT_LAG_8,
    l9.OCCURREDAT AS OCCURREDAT_LAG_9,
    l10.OCCURREDAT AS OCCURREDAT_LAG_10,
    l1.CORRECTNESS AS CORRECTNESS_LAG_1,
    l2.CORRECTNESS AS CORRECTNESS_LAG_2,
    l3.CORRECTNESS AS CORRECTNESS_LAG_3,
    l4.CORRECTNESS AS CORRECTNESS_LAG_4,
    l5.CORRECTNESS AS CORRECTNESS_LAG_5,
    l6.CORRECTNESS AS CORRECTNESS_LAG_6,
    l7.CORRECTNESS AS CORRECTNESS_LAG_7,
    l8.CORRECTNESS AS CORRECTNESS_LAG_8,
    l9.CORRECTNESS AS CORRECTNESS_LAG_9,
    l10.CORRECTNESS AS CORRECTNESS_LAG_10,
    l1.DURATIONSECONDS AS DURATIONSECONDS_LAG_1,
    l2.DURATIONSECONDS AS DURATIONSECONDS_LAG_2,
    l3.DURATIONSECONDS AS DURATIONSECONDS_LAG_3,
    l4.DURATIONSECONDS AS DURATIONSECONDS_LAG_4,
    l5.DURATIONSECONDS AS DURATIONSECONDS_LAG_5,
    l6.DURATIONSECONDS AS DURATIONSECONDS_LAG_6,
    l7.DURATIONSECONDS AS DURATIONSECONDS_LAG_7,
    l8.DURATIONSECONDS AS DURATIONSECONDS_LAG_8,
    l9.DURATIONSECONDS AS DURATIONSECONDS_LAG_9,
    l10.DURATIONSECONDS AS DURATIONSECONDS_LAG_10,
    -- Future correctness
    f1.CORRECTNESS AS FUTURE_CORRECTNESS_LAG_1,
    f2.CORRECTNESS AS FUTURE_CORRECTNESS_LAG_2,
    f3.CORRECTNESS AS FUTURE_CORRECTNESS_LAG_3,
    f4.CORRECTNESS AS FUTURE_CORRECTNESS_LAG_4,
    f5.CORRECTNESS AS FUTURE_CORRECTNESS_LAG_5
FROM base_events b
-- Past joins
LEFT JOIN base_events l1 ON b.STUDENTID = l1.STUDENTID AND b.SKILL = l1.SKILL AND b.event_rank = l1.event_rank + 1
LEFT JOIN base_events l2 ON b.STUDENTID = l2.STUDENTID AND b.SKILL = l2.SKILL AND b.event_rank = l2.event_rank + 2
LEFT JOIN base_events l3 ON b.STUDENTID = l3.STUDENTID AND b.SKILL = l3.SKILL AND b.event_rank = l3.event_rank + 3
LEFT JOIN base_events l4 ON b.STUDENTID = l4.STUDENTID AND b.SKILL = l4.SKILL AND b.event_rank = l4.event_rank + 4
LEFT JOIN base_events l5 ON b.STUDENTID = l5.STUDENTID AND b.SKILL = l5.SKILL AND b.event_rank = l5.event_rank + 5
LEFT JOIN base_events l6 ON b.STUDENTID = l6.STUDENTID AND b.SKILL = l6.SKILL AND b.event_rank = l6.event_rank + 6
LEFT JOIN base_events l7 ON b.STUDENTID = l7.STUDENTID AND b.SKILL = l7.SKILL AND b.event_rank = l7.event_rank + 7
LEFT JOIN base_events l8 ON b.STUDENTID = l8.STUDENTID AND b.SKILL = l8.SKILL AND b.event_rank = l8.event_rank + 8
LEFT JOIN base_events l9 ON b.STUDENTID = l9.STUDENTID AND b.SKILL = l9.SKILL AND b.event_rank = l9.event_rank + 9
LEFT JOIN base_events l10 ON b.STUDENTID = l10.STUDENTID AND b.SKILL = l10.SKILL AND b.event_rank = l10.event_rank + 10
-- Future joins
LEFT JOIN base_events f1 ON b.STUDENTID = f1.STUDENTID AND b.SKILL = f1.SKILL AND b.event_rank = f1.event_rank - 1
LEFT JOIN base_events f2 ON b.STUDENTID = f2.STUDENTID AND b.SKILL = f2.SKILL AND b.event_rank = f2.event_rank - 2
LEFT JOIN base_events f3 ON b.STUDENTID = f3.STUDENTID AND b.SKILL = f3.SKILL AND b.event_rank = f3.event_rank - 3
LEFT JOIN base_events f4 ON b.STUDENTID = f4.STUDENTID AND b.SKILL = f4.SKILL AND b.event_rank = f4.event_rank - 4
LEFT JOIN base_events f5 ON b.STUDENTID = f5.STUDENTID AND b.SKILL = f5.SKILL AND b.event_rank = f5.event_rank - 5
ORDER BY b.STUDENTID, b.SKILL, b.OCCURREDAT DESC, b.ANSWERPOSITIONINSESSION DESC
